# Astro Static Website — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the openlegalcommentary.ch static website using Astro, reading content from `../content/` at build time to produce pages for 8 Swiss federal laws and 4,694 articles.

**Architecture:** Astro static site in `site/` that reads YAML metadata and Markdown content from `../content/{law}/art-{NNN}/` at build time. Dynamic routes via `getStaticPaths()`. Zero JS framework — pure Astro components + vanilla CSS. Minimal client-side JS only for tab switching and theme toggle.

**Tech Stack:** Astro 4.x, Node.js 20+, js-yaml, gray-matter (optional), vanilla CSS with custom properties

**Content location:** `../content/` relative to `site/` (i.e., `/Users/jonashertner/Projects/openlegalcommentary/content/`)

// site/src/lib/content.ts
import fs from 'node:fs';
import path from 'node:path';
import yaml from 'js-yaml';

const CONTENT_ROOT = path.resolve(import.meta.dirname, '../../..', 'content');

export interface LayerMeta {
  last_generated: string;
  version: number;
  quality_score?: number;
  trigger?: string;
  last_reviewed?: string;
  total_decisions?: number;
  new_decisions_count?: number;
}

export interface ArticleMeta {
  law: string;
  article: number;
  article_suffix: string;
  title: string;
  sr_number: string;
  absatz_count: number;
  fedlex_url: string;
  lexfind_id?: number;
  lexfind_url: string;
  in_force_since: string;
  layers: Record<string, LayerMeta>;
}

export interface ArticleContent {
  meta: ArticleMeta;
  summary: string;
  doctrine: string;
  caselaw: string;
  /** Directory name, e.g., "art-001" or "art-006a" */
  dirName: string;
  /** URL slug, e.g., "art-1" or "art-6a" */
  slug: string;
  /** Available translations */
  translations: {
    summary?: { fr?: string; it?: string };
    doctrine?: { fr?: string; it?: string };
    caselaw?: { fr?: string; it?: string };
  };
}

export interface LawStats {
  totalArticles: number;
  completedArticles: number;
  /** Articles that have at least one non-placeholder layer */
  withContent: number;
}

/**
 * Convert directory name like "art-001" to URL slug like "art-1",
 * and "art-006a" to "art-6a".
 */
function dirNameToSlug(dirName: string): string {
  const match = dirName.match(/^art-(\d+)([a-z]*)$/);
  if (!match) return dirName;
  const num = parseInt(match[1], 10);
  const suffix = match[2] || '';
  return `art-${num}${suffix}`;
}

/**
 * Convert URL slug like "art-1" back to directory name like "art-001",
 * and "art-6a" to "art-006a".
 */
export function slugToDirName(slug: string): string {
  const match = slug.match(/^art-(\d+)([a-z]*)$/);
  if (!match) return slug;
  const num = match[1].padStart(3, '0');
  const suffix = match[2] || '';
  return `art-${num}${suffix}`;
}

function readFileIfExists(filePath: string): string {
  try {
    return fs.readFileSync(filePath, 'utf-8');
  } catch {
    return '';
  }
}

function isPlaceholder(content: string): boolean {
  const lines = content.trim().split('\n').filter((l) => l.trim().length > 0);
  return lines.length <= 2;
}

/**
 * Load metadata for a single article.
 */
export function loadArticleMeta(law: string, dirName: string): ArticleMeta | null {
  const metaPath = path.join(CONTENT_ROOT, law.toLowerCase(), dirName, 'meta.yaml');
  try {
    const raw = fs.readFileSync(metaPath, 'utf-8');
    return yaml.load(raw) as ArticleMeta;
  } catch {
    return null;
  }
}

/**
 * Load full content for a single article.
 */
export function loadArticle(law: string, dirName: string): ArticleContent | null {
  const artDir = path.join(CONTENT_ROOT, law.toLowerCase(), dirName);

  if (!fs.existsSync(artDir)) return null;

  const meta = loadArticleMeta(law, dirName);
  if (!meta) return null;

  const summary = readFileIfExists(path.join(artDir, 'summary.md'));
  const doctrine = readFileIfExists(path.join(artDir, 'doctrine.md'));
  const caselaw = readFileIfExists(path.join(artDir, 'caselaw.md'));

  const translations: ArticleContent['translations'] = {};

  // Check for French/Italian translations
  for (const layer of ['summary', 'doctrine', 'caselaw'] as const) {
    const fr = readFileIfExists(path.join(artDir, `${layer}.fr.md`));
    const it = readFileIfExists(path.join(artDir, `${layer}.it.md`));
    if (fr || it) {
      translations[layer] = {};
      if (fr) translations[layer]!.fr = fr;
      if (it) translations[layer]!.it = it;
    }
  }

  return {
    meta,
    summary,
    doctrine,
    caselaw,
    dirName,
    slug: dirNameToSlug(dirName),
    translations,
  };
}

/**
 * List all article directory names for a law, sorted.
 */
export function listArticleDirs(law: string): string[] {
  const lawDir = path.join(CONTENT_ROOT, law.toLowerCase());
  try {
    const entries = fs.readdirSync(lawDir, { withFileTypes: true });
    return entries
      .filter((e) => e.isDirectory() && e.name.startsWith('art-'))
      .map((e) => e.name)
      .sort((a, b) => {
        // Sort by number first, then suffix
        const matchA = a.match(/^art-(\d+)([a-z]*)$/);
        const matchB = b.match(/^art-(\d+)([a-z]*)$/);
        if (!matchA || !matchB) return a.localeCompare(b);
        const numA = parseInt(matchA[1], 10);
        const numB = parseInt(matchB[1], 10);
        if (numA !== numB) return numA - numB;
        return (matchA[2] || '').localeCompare(matchB[2] || '');
      });
  } catch {
    return [];
  }
}

/**
 * Load all articles for a law (metadata only for listing).
 */
export function listArticles(law: string): { meta: ArticleMeta; dirName: string; slug: string }[] {
  const dirs = listArticleDirs(law);
  const articles: { meta: ArticleMeta; dirName: string; slug: string }[] = [];

  for (const dirName of dirs) {
    const meta = loadArticleMeta(law, dirName);
    if (meta) {
      articles.push({ meta, dirName, slug: dirNameToSlug(dirName) });
    }
  }

  return articles;
}

/**
 * Compute statistics for a law.
 */
export function getLawStats(law: string): LawStats {
  const dirs = listArticleDirs(law);
  let withContent = 0;

  for (const dirName of dirs) {
    const artDir = path.join(CONTENT_ROOT, law.toLowerCase(), dirName);
    const summary = readFileIfExists(path.join(artDir, 'summary.md'));
    if (!isPlaceholder(summary)) {
      withContent++;
    }
  }

  return {
    totalArticles: dirs.length,
    completedArticles: withContent,
    withContent,
  };
}

/**
 * Get previous and next article for pagination.
 */
export function getArticleNav(
  law: string,
  currentDirName: string
): { prev: { slug: string; title: string } | null; next: { slug: string; title: string } | null } {
  const dirs = listArticleDirs(law);
  const idx = dirs.indexOf(currentDirName);

  let prev: { slug: string; title: string } | null = null;
  let next: { slug: string; title: string } | null = null;

  if (idx > 0) {
    const meta = loadArticleMeta(law, dirs[idx - 1]);
    if (meta) prev = { slug: dirNameToSlug(dirs[idx - 1]), title: meta.title };
  }

  if (idx >= 0 && idx < dirs.length - 1) {
    const meta = loadArticleMeta(law, dirs[idx + 1]);
    if (meta) next = { slug: dirNameToSlug(dirs[idx + 1]), title: meta.title };
  }

  return { prev, next };
}

import Base from '../../layouts/Base.astro';

export function getStaticPaths() {
  return LAWS.map((law) => ({
    params: { law: law.abbr.toLowerCase() },
  }));
}

const { law: lawParam } = Astro.params;
const lawInfo = getLawByAbbr(lawParam!);

if (!lawInfo) {
  return Astro.redirect('/');
}

const articles = listArticles(lawInfo.abbr);
const stats = getLawStats(lawInfo.abbr);

// Group articles by "hundred" for large laws, or show flat list
const useGrouping = articles.length > 50;

interface ArticleGroup {
  label: string;
  articles: typeof articles;
}

let groups: ArticleGroup[] = [];

if (useGrouping) {
  const grouped = new Map<number, typeof articles>();
  for (const art of articles) {
    const group = Math.floor((art.meta.article - 1) / 100);
    if (!grouped.has(group)) grouped.set(group, []);
    grouped.get(group)!.push(art);
  }
  groups = Array.from(grouped.entries())
    .sort(([a], [b]) => a - b)
    .map(([group, arts]) => {
      const start = group * 100 + 1;
      const end = (group + 1) * 100;
      return { label: `Art. ${start}–${end}`, articles: arts };
    });
} else {
  groups = [{ label: '', articles }];
}

function mdToHtml(md: string): string {
  if (!md.trim()) return '<p class="text-muted">Noch kein Inhalt verfuegbar.</p>';

  let html = md
    // Escape HTML
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    // Headers
    .replace(/^### (.+)$/gm, '<h3>$1</h3>')
    .replace(/^## (.+)$/gm, '<h2>$1</h2>')
    .replace(/^# (.+)$/gm, '<h1>$1</h1>')
    // Bold
    .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
    // Italic
    .replace(/\*(.+?)\*/g, '<em>$1</em>')
    // BGE references → link to opencaselaw.ch
    .replace(
      /\b(BGE\s+\d+\s+[IVX]+\s+\d+)/g,
      (match) => {
        const encoded = encodeURIComponent(match);
        return `<a href="https://opencaselaw.ch/search?q=${encoded}" target="_blank" rel="noopener">${match}</a>`;
      }
    )
    // Cross-references → internal links
    .replace(
      /(?:→|↔)\s*Art\.\s*(\d+)([a-z]*)\s+(BV|ZGB|OR|ZPO|StGB|StPO|SchKG|VwVG)/g,
      (match, num, suffix, law) => {
        const lawLower = law.toLowerCase();
        return `<a href="/${lawLower}/art-${parseInt(num)}${suffix}">${match}</a>`;
      }
    )
    // Blockquotes
    .replace(/^&gt; (.+)$/gm, '<blockquote><p>$1</p></blockquote>')
    // Paragraphs (lines separated by blank lines)
    .replace(/\n\n/g, '</p><p>')
    // Line breaks within paragraphs
    .replace(/\n/g, '<br/>');

  // Wrap in paragraph if not starting with a block element
  if (!html.startsWith('<h') && !html.startsWith('<blockquote') && !html.startsWith('<ul') && !html.startsWith('<ol')) {
    html = '<p>' + html + '</p>';
  }

  // Clean up empty paragraphs
  html = html.replace(/<p><\/p>/g, '');
  // Fix paragraphs around headers
  html = html.replace(/<p>(<h[1-3]>)/g, '$1');
  html = html.replace(/(<\/h[1-3]>)<\/p>/g, '$1');

  return html;
}

// site/src/pages/changelog.astro
import Base from '../layouts/Base.astro';
import Breadcrumb from '../components/Breadcrumb.astro';
import fs from 'node:fs';
import path from 'node:path';
import yaml from 'js-yaml';
import type { ArticleMeta, LayerMeta } from '../lib/content';

const CONTENT_ROOT = path.resolve(import.meta.dirname, '../../../content');

// Scan all articles for recently updated layers
interface ChangeEntry {
  date: string;
  law: string;
  article: number;
  suffix: string;
  layer: string;
  version: number;
  slug: string;
}

const changes: ChangeEntry[] = [];

const lawDirs = fs.readdirSync(CONTENT_ROOT, { withFileTypes: true })
  .filter((e) => e.isDirectory())
  .map((e) => e.name);

for (const lawDir of lawDirs) {
  const lawPath = path.join(CONTENT_ROOT, lawDir);
  const artDirs = fs.readdirSync(lawPath, { withFileTypes: true })
    .filter((e) => e.isDirectory() && e.name.startsWith('art-'));

  for (const artDir of artDirs) {
    const metaPath = path.join(lawPath, artDir.name, 'meta.yaml');
    try {
      const raw = fs.readFileSync(metaPath, 'utf-8');
      const meta = yaml.load(raw) as ArticleMeta;
      if (!meta.layers) continue;

      const match = artDir.name.match(/^art-(\d+)([a-z]*)$/);
      if (!match) continue;
      const num = parseInt(match[1], 10);
      const suffix = match[2] || '';
      const slug = `art-${num}${suffix}`;

      for (const [layerName, layerMeta] of Object.entries(meta.layers)) {
        const lm = layerMeta as LayerMeta;
        if (lm.last_generated) {
          changes.push({
            date: lm.last_generated,
            law: meta.law,
            article: meta.article,
            suffix: meta.article_suffix || '',
            layer: layerName,
            version: lm.version || 1,
            slug,
          });
        }
      }
    } catch {
      // Skip articles with invalid meta
    }
  }
}

// Sort by date descending
changes.sort((a, b) => b.date.localeCompare(a.date));

// Group by date
const byDate = new Map<string, ChangeEntry[]>();
for (const change of changes) {
  if (!byDate.has(change.date)) byDate.set(change.date, []);
  byDate.get(change.date)!.push(change);
}

const dates = Array.from(byDate.keys()).sort().reverse();

const layerLabels: Record<string, string> = {
  summary: 'Uebersicht',
  doctrine: 'Doktrin',
  caselaw: 'Rechtsprechung',
};

## Implementation Notes

### Architecture Decisions

1. **Content loading at build time via Node.js `fs`**: Astro runs in Node.js during build. We use `fs.readFileSync` to read YAML and Markdown files directly from `../content/`. No Astro content collections are used because the content format (separate `meta.yaml` + multiple `.md` files per directory) does not match Astro's expected content collection format.

2. **Simple Markdown rendering**: A basic regex-based Markdown-to-HTML converter is used inline rather than pulling in a full Markdown parser. This keeps dependencies minimal. The converter handles headers, bold, italic, blockquotes, and specifically adds hyperlinks for BGE references (to opencaselaw.ch) and cross-references (internal links). If richer Markdown features are needed later, `marked` or `markdown-it` can be added.

3. **URL structure**: Article slugs use un-padded numbers (`art-1`, not `art-001`) for cleaner URLs. The `slugToDirName` function converts back to zero-padded directory names for file system lookups.

4. **Dark mode**: Uses both `prefers-color-scheme` media query (auto) and a `data-theme` attribute (manual toggle). Theme preference is persisted in `localStorage` and applied before first paint via an inline script to prevent flash.

5. **Tab state in URL**: The `?layer=` query parameter is used for direct linking to specific tabs. This is updated via `history.replaceState` without page reload.

6. **Translation handling**: Translations are pre-rendered to HTML at build time and embedded as a JSON `<script>` block. Client-side JavaScript swaps the tab panel innerHTML when the language is changed. This avoids any API calls or dynamic loading.

7. **Build performance**: With 4,694 article pages, build time may be significant. Astro's static output mode handles this well. The `listArticleDirs` function reads directory listings once and caches via Astro's build system.
